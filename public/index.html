🟨 JavaScript / Node.js

const res = await fetch('/api/inbox', {
  method: 'POST',
  headers: { 'X-API-Key': 'xm_your_key' }
});
const { data: inbox } = await res.json();

// Get messages
const msgs = await fetch(
  `/api/inbox/${inbox.id}/messages`
).then(r => r.json());

🐍 Python

import requests

headers = {'X-API-Key': 'xm_your_key'}
inbox = requests.post(
    '/api/inbox', headers=headers
).json()['data']

msgs = requests.get(
    f'/api/inbox/{inbox["id"]}/messages'
).json()

🔵 cURL

# Create inbox
curl -X POST /api/inbox \
  -H "X-API-Key: xm_your_key"

# Get messages
curl /api/inbox/{id}/messages

# SSE real-time stream
curl -N /api/inbox/{id}/stream
